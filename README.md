class Solution {
    public static int findElementAtIndex( int key, int[] arr) {
        return arr[key];
    }

    public static void main(String[] args) {
        int[] arr = {10, 20, 30, 40, 50};
        int key = 2;
            System.out.println(findElementAtIndex(key, arr));
    }
}
